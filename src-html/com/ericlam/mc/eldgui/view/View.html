001package com.ericlam.mc.eldgui.view;
002
003import com.ericlam.mc.eld.services.ItemStackService;
004import com.ericlam.mc.eldgui.UIContext;
005import com.ericlam.mc.eldgui.UISession;
006import com.ericlam.mc.eldgui.model.Model;
007import org.bukkit.entity.Player;
008
009public abstract class View<T extends Model> {
010
011    protected final ItemStackService itemStackService;
012
013    public View(ItemStackService itemStackService) {
014        this.itemStackService = itemStackService;
015    }
016
017    public abstract T renderAndCreateModel(UISession session, UIContext context, Player player);
018
019    public abstract void onModelChanged(T model, UIContext context, Player player);
020
021    public void onResume(UISession session, UIContext context, Player player){}
022
023    public boolean persist() {
024        return false;
025    }
026
027}
